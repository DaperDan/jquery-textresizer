JavaScript/jQuery Code

<script type="text/javascript">
   jQuery(document).ready( function() {
      jQuery( "#textsizer-classnames a" ).textresizer({
         target: "#maincontent",
         type: "cssClass",
         sizes: [ 
            "small-text", 
            "medium-text", 
            "large-text", 
            "larger-text" 
         ],
         selectedIndex: 1
      });
   });
</script>
